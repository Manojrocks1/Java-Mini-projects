package Day2_StudentGradeCalculator.src;

import java.util.*;
import java.io.*;

// -----------------------------
// ğŸ“ Student Model Class
// -----------------------------
class Student {
    String studentId, name;
    int[] marks;           // Array to store subject marks
    int total;             // Sum of marks
    double average;        // Average of marks
    String grade;          // Grade based on average

    // âš™ï¸ Constructor: initializes all parameters and evaluates scores
    public Student(String studentId, String name, int[] marks) {
        this.studentId = studentId;
        this.name = name;
        this.marks = marks;
        calculateTotal();
        calculateAverage();
        calculateGrade();
    }

    // âœ… Calculate total marks
    void calculateTotal() {
        total = Arrays.stream(marks).sum(); // Java 8+ stream for summing array
    }

    // âœ… Compute average score
    void calculateAverage() {
        average = total / (double) marks.length; // Cast to double for precision
    }

    // âœ… Determine grade based on average
    void calculateGrade() {
        if (average >= 90) grade = "A+";
        else if (average >= 80) grade = "A";
        else if (average >= 70) grade = "B";
        else if (average >= 60) grade = "C";
        else grade = "D";
    }

    // ğŸ–¨ï¸ Display report on console
    void printReport() {
        System.out.printf("""
                \n--- STUDENT REPORT ---
                ID: %s
                Name: %s
                Total: %d
                Average: %.2f
                Grade: %s
                """, studentId, name, total, average, grade);
    }

    // ğŸ’¾ Format report as a string for saving to file
    String toFileString() {
        return String.format("""
                --- STUDENT REPORT ---
                ID: %s
                Name: %s
                Total: %d
                Average: %.2f
                Grade: %s
                ----------------------

                """, studentId, name, total, average, grade);
    }
}

// -----------------------------
// ğŸ“˜ Main Application Class
// -----------------------------
public class AdvancedStudentReportCard {
    
    // ğŸ§¾ Configuration: Total number of subjects
    static final int SUBJECTS = 5;

    // ğŸ¯ Scanner for user input and list to hold all students
    static Scanner sc = new Scanner(System.in);
    static List<Student> students = new ArrayList<>();

    // ğŸš€ Main method: program entry point
    public static void main(String[] args) {
        System.out.println("ğŸ“˜ Welcome to the Student Report Card System");

        while (true) {
            // Input Student ID or exit condition
            System.out.print("\nEnter Student ID (or type 'exit' to finish): ");
            String id = sc.nextLine();
            if (id.equalsIgnoreCase("exit")) break;

            // Check if ID is already used
            if (isDuplicateId(id)) {
                System.out.println("âš ï¸ Student ID already exists.");
                continue;
            }

            // Input student name
            System.out.print("Enter Student Name: ");
            String name = sc.nextLine();

            // Input marks for defined subjects with validation
            int[] marks = new int[SUBJECTS];
            for (int i = 0; i < SUBJECTS; i++) {
                marks[i] = inputValidMark(i + 1);
            }

            // Add validated and constructed Student object to list
            students.add(new Student(id, name, marks));
            System.out.println("âœ… Student added.");
        }

        // Display report and save to a file
        generateReports();
    }

    // ğŸ›¡ï¸ Method to validate mark input between 0 and 100
    static int inputValidMark(int subjectNo) {
        while (true) {
            System.out.print("Marks for Subject " + subjectNo + ": ");
            try {
                int mark = Integer.parseInt(sc.nextLine());
                if (mark >= 0 && mark <= 100) return mark;
            } catch (NumberFormatException ignored) { }
            System.out.println("âŒ Enter a number between 0 and 100.");
        }
    }

    // ğŸ” Check if Student ID already exists in list (case-insensitive)
    static boolean isDuplicateId(String id) {
        return students.stream().anyMatch(s -> s.studentId.equalsIgnoreCase(id));
    }

    // ğŸ§¾ Generate and print all reports + save to 'Student_reports.txt'
    static void generateReports() {
        if (students.isEmpty()) {
            System.out.println("No student records to display.");
            return;
        }

        // Try-with-resources: auto-closes FileWriter
        try (FileWriter fw = new FileWriter("Student_reports.txt")) {
            for (Student s : students) {
                s.printReport();            // Print report to console
                fw.write(s.toFileString()); // Write report to file
            }
            System.out.println("\nğŸ“ All reports saved to 'Student_reports.txt'");
        } catch (IOException e) {
            System.out.println("âš ï¸ Error saving file: " + e.getMessage());
        }
    }
}
